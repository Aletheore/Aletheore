# Defensive Infrastructure (Phase 6) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give dead-code detection a real (if narrow) production signal instead of pure static analysis, make managed-audit reports independently verifiable after the fact, and surface a PR's blast radius against recent production incidents - all without inventing infrastructure this codebase doesn't actually have.

**Architecture:** Three independent pieces, each grounded in infrastructure that already exists rather than assuming a telemetry pipeline this project doesn't have. (1) Stale-endpoint detection cross-references the deterministic endpoint list against `endpoint_health` history (the synthetic health-check sweep from Phase 4) - an endpoint with several checks and zero successful ones in the window is flagged, exposed via the existing dashboard health JSON. (2) Signed audit records persist each managed-audit report's text alongside an Ed25519 signature over its content hash, with a public (token-gated, not ID-gated) verification endpoint that confirms integrity without republishing the report's contents. (3) A regression-risk check run reuses `endpoint_health`'s incident history to flag - never block - a PR that touches a handler with recent reachability incidents, as a separate, clearly-advisory GitHub Check Run.

**Tech Stack:** Python 3.12, `cryptography` (Ed25519, already a dependency), psycopg (sync, scan_worker), asyncpg (async, app_server), pytest + pytest-asyncio.

## Global Constraints

- No piece in this plan assumes real production request-level telemetry exists. The only "telemetry" available is the synthetic health-check sweep (`endpoint_health`, from the prior phase) - every signal here is derived from that table, not invented.
- The regression-risk check run is advisory only - it must use GitHub's `"neutral"` conclusion and a check-run **name** distinct from `"Aletheore secrets check"` (`"Aletheore regression risk"`), so it can never affect a branch-protection rule scoped to the secrets check, and never blocks a merge.
- Audit-report verification is keyed by a random 32-byte token generated at write time, never by the row's sequential ID - report text can contain sensitive findings (secret locations, vulnerable dependency names) about a private repo, and a sequential ID would be enumerable.
- The verify endpoint returns integrity metadata (hash, signed-at, verified boolean, repo name) - it never re-serves the report's full text. Anyone who needs the text already received it via the PR comment or API response that generated it.
- Managed audits and Flash review's own model-call paths are untouched by this plan - Task 2 only adds persistence and signing *after* a report is already generated by existing code.

```python
MIN_CHECKS_FOR_STALE_CONFIDENCE = 5


def find_stale_endpoints(
    endpoints: list[dict], health_summary: dict[tuple[str, str], dict]
) -> list[dict]:
    """Endpoints with several checks in the window and zero of them
    reachable - a much stronger dead-code signal than static import-graph
    reachability alone, since it reflects the endpoint's real, live URL.
    """
    stale = []
    for endpoint in endpoints:
        key = (endpoint.get("method"), endpoint.get("path"))
        summary = health_summary.get(key)
        if summary is None:
            continue
        if summary["ever_reachable"] or summary["check_count"] < MIN_CHECKS_FOR_STALE_CONFIDENCE:
            continue
        stale.append(
            {
                "method": endpoint.get("method"),
                "path": endpoint.get("path"),
                "file": endpoint.get("file"),
                "line": endpoint.get("line"),
                "check_count": summary["check_count"],
            }
        )
    return stale
```

```python
async def get_endpoint_health_summary_since(
    pool: asyncpg.Pool, installation_id: int, repo_full_name: str, since: datetime
) -> dict[tuple[str, str], dict]:
    rows = await pool.fetch(
        """
        SELECT endpoint_method, endpoint_path, bool_or(reachable) AS ever_reachable, count(*) AS check_count
        FROM endpoint_health
        WHERE installation_id = $1 AND repo_full_name = $2 AND checked_at >= $3
        GROUP BY endpoint_method, endpoint_path
        """,
        installation_id,
        repo_full_name,
        since,
    )
    return {
        (row["endpoint_method"], row["endpoint_path"]): {
            "ever_reachable": row["ever_reachable"],
            "check_count": row["check_count"],
        }
        for row in rows
    }
```

```python
"""Ed25519 signing for managed-audit reports.

A report's signature covers content_hash(report_text), not the raw text
itself - the signed payload is always the sha256 hex digest, so callers
never need to worry about newline/encoding drift between signing and
verification.
"""

import hashlib

from cryptography.exceptions import InvalidSignature
from cryptography.hazmat.primitives.asymmetric.ed25519 import Ed25519PrivateKey, Ed25519PublicKey


def content_hash(report_text: str) -> str:
    return hashlib.sha256(report_text.encode()).hexdigest()


def sign_report(report_text: str, private_key_hex: str) -> str:
    private_key = Ed25519PrivateKey.from_private_bytes(bytes.fromhex(private_key_hex))
    signature = private_key.sign(content_hash(report_text).encode())
    return signature.hex()


def verify_report(report_text: str, signature_hex: str, public_key_hex: str) -> bool:
    try:
        public_key = Ed25519PublicKey.from_public_bytes(bytes.fromhex(public_key_hex))
        public_key.verify(bytes.fromhex(signature_hex), content_hash(report_text).encode())
        return True
    except (InvalidSignature, ValueError):
        return False


def public_key_hex_from_private(private_key_hex: str) -> str:
    private_key = Ed25519PrivateKey.from_private_bytes(bytes.fromhex(private_key_hex))
    return private_key.public_key().public_bytes_raw().hex()
```

```sql
-- Verification is keyed by a random token, never the sequential id -
-- report_text can contain sensitive findings about a private repo, and a
-- sequential id would be enumerable by anyone hitting the verify endpoint.
CREATE TABLE IF NOT EXISTS audit_reports (
    id                  BIGSERIAL PRIMARY KEY,
    installation_id     BIGINT NOT NULL REFERENCES installations(installation_id) ON DELETE CASCADE,
    repo_full_name      TEXT NOT NULL,
    verification_token  TEXT NOT NULL UNIQUE,
    report_text         TEXT NOT NULL,
    content_hash        TEXT NOT NULL,
    signature           TEXT NOT NULL,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS audit_reports_token_lookup ON audit_reports (verification_token);
```

```python
    changed_files: list[str], evidence: dict, incidents: list[dict]
) -> list[dict]:
    incident_by_key = {(i["endpoint_method"], i["endpoint_path"]): i for i in incidents}
    endpoints = evidence.get("repository", {}).get("api_endpoints", {}).get("endpoints", [])
    changed = set(changed_files)
    touched = []
    for endpoint in endpoints:
        if endpoint.get("file") not in changed:
            continue
        key = (endpoint.get("method"), endpoint.get("path"))
        incident = incident_by_key.get(key)
        if incident is None:
            continue
        touched.append(
            {
                "method": endpoint.get("method"),
                "path": endpoint.get("path"),
                "file": endpoint.get("file"),
                "line": endpoint.get("line"),
                "incident_count": incident["incident_count"],
                "last_incident_at": incident["last_incident_at"],
            }
        )
    return touched
```

```python
def _maybe_create_regression_risk_check_run(
    client: httpx.Client,
    token: str,
    repo_full_name: str,
    head_sha: str,
    installation_id: int,
    evidence: dict,
    changed_files: list[str],
) -> None:
    settings = get_settings()
    installation = get_installation_row(settings.database_url, installation_id)
    if installation is None or installation["plan"] == "free":
        return

    since = datetime.now(timezone.utc) - timedelta(days=REGRESSION_FENCE_WINDOW_DAYS)
    incidents = list_recent_endpoint_incidents(settings.database_url, installation_id, repo_full_name, since)
    if not incidents:
        return

    touched = find_touched_incident_endpoints(changed_files, evidence, incidents)
    if not touched:
        return

    lines = []
    for item in touched:
        location = (
            f" - handled by {item['file']}:{item['line']}"
            if item.get("file") and item.get("line") is not None
            else ""
        )
        lines.append(
            f"- `{item['method']} {item['path']}`{location}: "
            f"{item['incident_count']} reachability incident(s) in the last {REGRESSION_FENCE_WINDOW_DAYS} days"
        )
    summary = "This PR touches a handler with recent production reachability incidents:\n" + "\n".join(lines)
    create_check_run(client, token, repo_full_name, head_sha, "neutral", summary, name="Aletheore regression risk")
```
